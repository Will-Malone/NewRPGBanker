'use strict';

const babylon = require('@babel/parser');
const options = require('./tsOptions');

/** 
 * Doesn't accept custom options because babylon should be used directly in 
 * that case. 
 */
module.exports = function() {
  return {
    parse(code) {
      return babylon.parse(code, options);
    },
  };
};
