#import <React/RCTVirtualTextShadowView.h>

#import <React/RCTShadowView+Layout.h>
#import <yoga/Yoga.h>

#import <React/RCTRawTextShadowView.h>

@implementation RCTVirtualTextShadowView {
  BOOL _isLayoutDirty;
}

#pragma mark - Layout

- (void)dirtyLayout
{
  [super dirtyLayout];

  if (_isLayoutDirty) {
    return;
  }
  _isLayoutDirty = YES;

  [self.superview dirtyLayout];
}

- (void)clearLayout
{
  _isLayoutDirty = NO;
}

@end
