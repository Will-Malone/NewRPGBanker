#include "YGJNIVanilla.h"
#include "common.h"

using namespace facebook::yoga;

jint JNI_OnLoad(JavaVM* vm, void*) {
  JNIEnv* env;
  jint ret = vanillajni::ensureInitialized(&env, vm);
  YGJNIVanilla::registerNatives(env);
  return ret;
}
