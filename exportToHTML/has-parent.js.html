/** 
 * @private 
 * find if a node is in the given parent 
 * @method hasParent 
 * @param {HTMLElement} node 
 * @param {HTMLElement} parent 
 * @return {Boolean} found 
 */
export default function hasParent(node, parent) {
  while (node) {
    if (node === parent) {
      return true;
    }
    node = node.parentNode;
  }
  return false;
}
