#import "RCTViewUtils.h"

#import "UIView+React.h"

UIEdgeInsets RCTContentInsets(UIView *view)
{
  while (view) {
    UIViewController *controller = view.reactViewController;
    if (controller) {
      return controller.view.safeAreaInsets;
    }
    view = view.superview;
  }
  return UIEdgeInsetsZero;
}
