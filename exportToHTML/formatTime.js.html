function formatTime(time, prefixPower = -3, padLeftLength = 0) {
  const prefixes = ['n', 'Î¼', 'm', ''];
  const prefixIndex = Math.max(
    0,
    Math.min(
      Math.trunc(prefixPower / 3) + prefixes.length - 1,
