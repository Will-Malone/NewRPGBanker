    get name(): string;
    get NativeGestureClass(): PanRecognizerStatic;
    getHammerConfig(): {
        direction: number;
        pointers: number | undefined;
    };
    getState(type: keyof typeof EventMap): State;
    getDirection(): number;
    getConfig(): Partial<{
        enabled: boolean;
        minPointers: number;
        maxPointers: number;
        minDist: number;
        minDistSq: number;
        minVelocity: number;
        minVelocitySq: number;

        success?: undefined;

        failed?: undefined;
    }; 
}
