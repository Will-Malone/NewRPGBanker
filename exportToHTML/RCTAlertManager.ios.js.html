import type {Args} from './NativeAlertManager';

import NativeAlertManager from './NativeAlertManager';

module.exports = {
  alertWithArgs(
    args: Args,
    callback: (id: number, value: string) => void,
  ): void {
    if (NativeAlertManager == null) {
      return;
    }
    NativeAlertManager.alertWithArgs(args, callback);
  },
};
