'use strict';

import type {GraphId} from '../lib/getGraphId';

class GraphNotFoundError extends Error {
  graphId: GraphId;

  constructor(graphId: GraphId) {
    super(`The graph \`${graphId}\` was not found.`);
    this.graphId = graphId;
  }
}

module.exports = GraphNotFoundError;
