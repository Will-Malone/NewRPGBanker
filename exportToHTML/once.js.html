/** 
 * This replicates lodash's once functionality for our purposes. 
 */
module.exports = function(func) {
  let called = false;
  let result;
  return function(...args) {
    if (called) {
      return result;
    }
    called = true;
    return result = func.apply(this, args);
  };
};
