'use strict';

type Point = {
  x: ?number,
  y: ?number,
  ...
};

const dummyPoint = {x: undefined, y: undefined};

const pointsDiffer = function (one: ?Point, two: ?Point): boolean {
  one = one || dummyPoint;
  two = two || dummyPoint;
  return one !== two && (one.x !== two.x || one.y !== two.y);
};

module.exports = pointsDiffer;
